73  ZMap<std::string, SharedResource<RType>>& map) {
74  auto itr = map.find(path);
75  if (itr != map.end()) {
76  if (itr->second.isValid()) {
77  itr->second.increaseReference();

79  } else {

84  SharedResource<RType> &res = map[path];
85  res.load(path);
86  if (res.isValid()) {

89  } else {

91  }
92  } else {
93  Logger::getInstance().write(
94  "During loading: " + path
95  + " via ResouceCache -> Reason: File does not exits",
96  Logger::MessageType::Error);
97  map[path].setValid(false);

99  }

112  Logger::getInstance().write(

114  map.erase(itr);
115  }
116  }
