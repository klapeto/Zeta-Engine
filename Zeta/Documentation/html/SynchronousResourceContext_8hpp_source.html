204  template<typename RType>
205  const RType& getResource(const std::string& path,
206  std::unordered_map<std::string, SharedResource<RType>>& map,
207  std::mutex& mutex) {
208  std::unique_lock<std::mutex> lock(mutex);
209  auto itr = map.find(path);
210  if (itr != map.end()) {
211  if (itr->second.isValid()) {
212  itr->second.increaseReference();
213  return itr->second.getInstance();
214  } else {
215  return NullReference<RType>::getNull();
216  }
217  } else {
218  if (FileLoader::fileExists(path.c_str())) {
219  SharedResource<RType> &res = map[path];
220  res.load(path);
221  if (res.isValid()) {
222  res.increaseReference();
223  return res.getInstance();
224  } else {
225  return NullReference<RType>::getNull();
226  }
227  } else {
228  Logger::getInstance().write(
229  "During loading: " + path
230  + " Reason: File does not exits",
231  Logger::MessageType::Error);
232  map[path].setValid(false);
233  return NullReference<RType>::getNull();
234  }
235  }
236  return NullReference<RType>::getNull();
237  }

239  template<typename RType>
240  void releaseMappedResource(const std::string& path,
241  std::unordered_map<std::string, SharedResource<RType>>& map,
242  std::mutex& mutex) {
243  std::unique_lock<std::mutex> lock(mutex);
244  auto itr = map.find(path);
245  if (itr != map.end()) {
246  if (itr->second.isValid()) {
247  itr->second.decreaseReference();
248  if (itr->second.getReferenceCount() <= 0) {
249  Logger::getInstance().write("Releasing Resource: " + path);
250  map.erase(itr);
251  }
252  }
253  }
254  }

256  const Bitmap& getBitmapImpl(const std::string& path) {
257  std::unique_lock<std::mutex> lock(bitmapMutex);
258  auto itr = bitmaps.find(path);
259  if (itr != bitmaps.end()) {
260  if (itr->second.isValid()) {
261  itr->second.increaseReference();
262  return itr->second.getInstance();
263  } else {
264  return NullReference<BitmapT>::getNull();
265  }
266  } else {
267  if (Display::isRenderingThread()) {
268  if (FileLoader::fileExists(path.c_str())) {
269  SharedResource<BitmapT> &res = bitmaps[path];
270  res.load(path);
271  if (res.isValid()) {
272  res.increaseReference();
273  return res.getInstance();
274  } else {
275  return NullReference<BitmapT>::getNull();
276  }
277  } else {
278  Logger::getInstance().write(
279  "During loading: " + path
280  + " Reason: File does not exits",
281  Logger::MessageType::Error);
282  bitmaps[path].setValid(false);
283  return NullReference<BitmapT>::getNull();
284  }
285  } else {
286  lock.unlock();
287  BitmapLoadRequest request(path);
288  System::getInstance().getRenderingContext().addOperationToQueue(
289  request);
290  request.waitToBeCompleted();
291  return request.getBitmap();
292  }
293  }
